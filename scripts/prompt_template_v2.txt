## In-context example: Rocket Launcher in the v2 strict profile

```javascript
'use strict';

const { createInstance } = require('@cognitive-fab/sam-pattern');

const instance = createInstance({ strict: true, hasAsyncActions: false });

// status: 'ready' | 'counting' | 'aborted' | 'launched'
const INITIAL_STATE = { counter: 10, status: 'ready' };

const control = instance({
  initialState: { ...INITIAL_STATE },
  component: {
    modelShape: {
      counter: { type: 'number' },
      status: { type: 'string' },
    },
    actions: {
      START: { action: (data = {}) => ({ ...data }), schema: {}, domain: [{}] },
      DECREMENT: { action: (data = {}) => ({ ...data }), schema: {}, domain: [{}] },
      ABORT: { action: (data = {}) => ({ ...data }), schema: {}, domain: [{}] },
    },
    acceptors: {
      START: (model) => (proposal, { reject }) => {
        if (model.status !== 'ready') return reject('start-only-when-ready');
        model.status = 'counting';
      },
      DECREMENT: (model) => (proposal, { reject }) => {
        if (model.status !== 'counting') return reject('not-counting');
        model.counter -= 1;
        if (model.counter === 0) model.status = 'launched';
      },
      ABORT: (model) => (proposal, { reject }) => {
        if (model.status !== 'counting') return reject('nothing-to-abort');
        model.status = 'aborted';
      },
    },
    reactors: [],
  },
});

const { intents } = control;

const getState = () => instance({}).getState();
const setState = (snapshot) => { instance({}).setState(snapshot); };

// NOTE: do not use instance({}).state() anywhere — getState() is the safe
// snapshot accessor (a model data key like 'state' would return data, not the
// method; see sam-lib #29). init() is EXACTLY this one call:
const init = () => { setState(INITIAL_STATE); };

const actions = {
  START: (data = {}) => intents.START(data),
  DECREMENT: (data = {}) => intents.DECREMENT(data),
  ABORT: (data = {}) => intents.ABORT(data),
};

module.exports = { instance, init, actions, getState, setState };
```

(The rocket's state is flat, so its acceptors assign model keys directly. When
a state key holds a nested map/object, commit each change with the top-level
write `model.<key> = { ...model.<key>, [k]: updated }` instead.)
